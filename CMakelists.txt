project(setup)
cmake_minimum_required(VERSION 3.11)
# Require modern C++
set(CMAKE_CXX_STANDARD 14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

#### Setup Directories ####
# Set the output directory to the highest relative folder
SET(CMAKE_RUNTIME_OUTPUT_DIRECTORY "${CMAKE_SOURCE_DIR}/GameBuild") # Where the executable is output

#### Add External Dependencies ####
add_subdirectory("lib/SFML")
set(SFML_INCS "lib/SFML/include")
link_directories("${CMAKE_BINARY_DIR}/lib/SFML/lib")

#### Practical 1 ####
add_executable(FixAllShop "src/main.cpp" "src/State.h" "src/MenuState.h" "src/MenuState.cpp" "src/GameState.h" "src/GameState.cpp" "src/OrbScene.h" "src/OrbScene.cpp" "src/DoorScene.h" "src/DoorScene.cpp" "src/SpellScene.h" "src/SpellScene.cpp" "src/GameVariables.h")
target_include_directories(FixAllShop PRIVATE ${SFML_INCS})
target_link_libraries(FixAllShop sfml-graphics)
